mobilize = {
  if = {
    limit = {
      check_variable = {
        var = num_of_battalions
        value = battalions_force_limit
        compare = lesser_than
      }
    }
    set_country_flag = mobilized
    set_temp_variable = {
      var = mobilizing_force
      value = battalions_force_limit
    }
    subtract_from_temp_variable = {
      var = mobilizing_force
      value = num_of_battalions
    }
    multiply_temp_variable = {
      var = mobilizing_force
      value = war_support
    }
    add_to_temp_variable = {
      var = mobilizing_force
      value = 1
    }
    
    while_loop_effect = {
      limit = {
        check_variable = {
          var = mobilizing_force
          value = 0
          compare = greater_than
        }
      }
      random_owned_state = {
        limit = {
          OR = {
            has_core_of = PREV
            is_capital = yes
          }
        }
        random_list = {
          0 = {
            modifier = {
              add = 40
              has_state_modifier = climate_arid
            }
            modifier = {
              add = 30
              has_state_modifier = climate_arid_coast
            }
            modifier = {
              add = 30
              has_state_modifier = climate_arid_highlands
            }
            modifier = {
              add = 30
              has_state_modifier = climate_arid
            }
            mobilize_camel_archers = yes
          }
          0 = {
            modifier = {
              add = 30
              has_state_modifier = climate_arid
            }
            modifier = {
              add = 40
              has_state_modifier = climate_arid_coast
            }
            modifier = {
              add = 40
              has_state_modifier = climate_arid_highlands
            }
            modifier = {
              add = 40
              has_state_modifier = climate_arid
            }
            mobilize_camel_warriors = yes
          }
          15 = {
            modifier = {
              add = 15
              PREV = { has_idea = reform_tier_1_nomadism }
            }
            mobilize_tribal_cavalry = yes
          }
          60 = {
            modifier = {
              add = -15
              PREV = { has_idea = reform_tier_1_nomadism }
            }
            mobilize_tribal_warriors = yes
          }
          25 = {
            mobilize_tribal_archers = yes
          }
        }
      }
    }
  }
}

    limit = {
      PREV = {
        check_variable = {
          var = mobilizing_force
          value = 6
          compare = greater_than
        }
      }
    }
    create_unit = {
      division = "name = \"Large Camel Archer Levy\" division_template = \"Large Camel Archer Levy\" start_experience_factor = 0 start_equipment_factor = 1.0"
      owner = PREV
    }
    PREV = {
      subtract_from_temp_variable = {
        var = mobilizing_force
        value = 3
