is_non_player_tag = {
	custom_trigger_tooltip = {
		tooltip = is_non_player_tag_tt
		if = {
			limit = {
				OR = {
					tag = TES
					tag = EUR
					tag = ENG
					tag = FRA
					tag = SPR
					tag = GER
					tag = ITA
					tag = AUS
					tag = DEN
					tag = SWE
					tag = RUS
					tag = USA
					tag = CSA
					tag = BRA
					tag = HOL
				}
			}
			always = yes
		}
		else = {
			always = no
		}
	}
}

is_player_tag = {
	custom_trigger_tooltip = {
		tooltip = is_player_tag_tt
		if = {
			limit = {
				OR = {
					tag = TES
					tag = EUR
				}
			}
			always = no
		}
		else = {
			always = yes
		}
	}
}

is_great_power = {
	check_variable = {
		prestige_rank < 9
		tooltip = prestige_rank_top_8
	}
	hidden_trigger = {
		has_variable = prestige_rank
		is_subject = no
	}
}

is_secondary_power = {
	check_variable = {
		prestige_rank < 17
		tooltip = prestige_rank_secondary
	}
	hidden_trigger = {
		check_variable = {
			prestige_rank > 8
		}
		is_subject = no
	}
}

is_minor_power = {
	check_variable = {
		prestige_rank < 25
		tooltip = prestige_rank_minor
	}
	hidden_trigger = {
		NOT = { is_great_power = yes }
		NOT = { is_secondary_power = yes }
	}
}

is_irrelevant_power = {
	check_variable = {
		prestige_rank > 24
		tooltip = prestige_rank_irrelevant
	}
}

toto_enabled_autonomy = {
	if = {
		limit = {
			OR = {
				has_dlc = "Together for Victory"
				has_dlc = "Death or Dishonor"
				has_dlc = "Waking the Tiger"
			}
		}
		always = yes
	}
	else = {
		always = no
	}
}
