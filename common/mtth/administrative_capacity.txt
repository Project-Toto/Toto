administrative_capacity_used = {
	base = 0
	modifier = {
		set_temp_variable = {
			var = administrative_capacity_used_temp
			value = 0			
		}
		for_each_scope_loop = {
			array = owned_states
			set_temp_variable = {
				var = modifier@local_administrative_capacity_temp
				value = local_administrative_capacity
			}
			if = {
				limit = {
					NOT = { is_core_of = PREV }
				}
				mutiply_temp_variable = {
					var = local_administrative_capacity_temp
					value = 0.4
				}
			}
			add_to_temp_variable = {
				var = administrative_capacity_used_temp
				value = local_administrative_capacity_used_temp
			}
		}
		add = administrative_capacity_used_temp
	}
}
administrative_overcapacity = {
	base = 0
	modifier = {
		set_temp_variable = {
			var = administrative_overcapacity_temp
			value = administrative_capacity_used
		}
		divide_temp_variable = {
			var = administrative_overcapacity_temp
			value = modifier@administrative_capacity
		}
		subtract_from_temp_variable = {
			var = administrative_overcapacity_temp
			value = 1
		}
		clamp_temp_variable = {
			var = administrative_overcapacity_temp
			min = 0
			max = 5
		}
		add = administrative_overcapacity_temp
	}
}
